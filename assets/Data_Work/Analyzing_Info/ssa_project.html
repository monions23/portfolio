# Read all files in the range, put each one into a Dataframe
# Make sure the DataFrame has the corresponding year in a new column
# As they are created in a loop, put DataFrames in an array, then use concat to make a single large DataFrame
chosen_dataframes = []
columns = ['name', 'sex', 'num_births']

for year in chosen_years:
    df = pd.read_csv('names/yob%d.txt' % year, names = columns)
    df['year'] = year
    chosen_dataframes.append(df)

all_names = pd.concat(chosen_dataframes, ignore_index = True)
