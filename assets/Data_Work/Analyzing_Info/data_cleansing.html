# Remove outliers on overall_score column using std (standard deviation) method

# Find outliers
lower_limit = df['overall_score'].mean() - (3 * df['overall_score'].std())
upper_limit = df['overall_score'].mean() + (3 * df['overall_score'].std())

# Keep rows with values in range
df = df[(df['overall_score'] > lower_limit) & (df['overall_score'] < upper_limit)]
